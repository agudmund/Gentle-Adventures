# Gentle Adventures — runtime dependencies
# "it works for me / it works for you / done" — no version pins; Python is the
# floor, pip resolves the rest.
#
# Family packages (pretty_widgets / shared_braincell / intricate_log) are
# editable installs from their sibling repos on the Desktop, not PyPI:
#   pip install -e "<desktop>/Pretty-Widgets"
#   pip install -e "<desktop>/Shared-Braincell"
#   pip install -e "<desktop>/intricate-log"      # maturin/pyo3 — builds the Rust .pyd

# ── Runtime: GUI ──────────────────────────────────────────────────────────
PySide6                 # Qt bindings — the frameless window, titlebar, scene view

# ── Runtime: shared family infrastructure ────────────────────────────────
pretty_widgets          # titlebar combo/button/tooltip, the shared Theme, Chandler42 fonts
shared_braincell        # settings resolution + the logger adapter onto intricate_log
intricate_log           # the family Rust logger backend — lock-free ring buffer,
                        # timestamped-per-run files, retention. Soft-falls to a
                        # pure-Python sink if the .pyd is absent; GA falls back further
                        # to a stdlib console+file logger if the whole spine is missing.
                        # Preferred, not hard-required — but a proper family member ships it.

# ── Standard library covers the rest ─────────────────────────────────────
# Gemini image generation goes over urllib + base64 + json (no requests).
# The NPU "ship's engine" probe shells out to PowerShell (no wmi/pywin32).
# TOML loads via stdlib tomllib on 3.11+; older Pythons need tomli:
tomli; python_version < "3.11"
